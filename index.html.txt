  <script>
    const questions = [
      { q: "O que significa automação no trabalho?", o: ["Substituição total de pessoas em todas as funções", "Uso de tecnologia para executar tarefas com menos intervenção humana", "Trabalho manual sem ferramentas digitais", "Apenas uso de robôs industriais"], c: 1 },
      { q: "Qual área costuma ser bastante impactada pela automação?", o: ["Tarefas repetitivas", "Somente artes visuais", "Apenas esportes", "Somente música"], c: 0 },
      { q: "Um benefício comum da automação é:", o: ["Redução de produtividade", "Mais erros manuais", "Aumento de eficiência", "Fim de toda necessidade humana"], c: 2 },
      { q: "Qual exemplo mostra automação no dia a dia?", o: ["Caixa eletrônico", "Lápis e papel", "Caderno de anotações", "Relógio analógico"], c: 0 },
      { q: "A automação pode criar novos empregos?", o: ["Não, nunca cria", "Sim, especialmente em tecnologia e manutenção", "Só em profissões antigas", "Somente em agricultura"], c: 1 },
      { q: "Qual habilidade tende a ser mais importante com a automação?", o: ["Decorar números aleatórios", "Pensamento crítico", "Escrever devagar", "Evitar tecnologia"], c: 1 },
      { q: "A automação substitui melhor tarefas:", o: ["Criativas e subjetivas", "Repetitivas e padronizadas", "Que exigem empatia", "Que exigem julgamento humano complexo"], c: 1 },
      { q: "Um risco da automação é:", o: ["Eliminar a necessidade de atualização profissional", "Aumentar a demanda por novas habilidades", "Desaparecimento de tarefas repetitivas", "Deslocamento de alguns empregos"], c: 3 },
      { q: "Qual profissional pode ser muito importante na automação?", o: ["Técnico em manutenção", "Somente poeta", "Somente pintor", "Somente jornalista"], c: 0 },
      { q: "A melhor postura diante da automação é:", o: ["Ignorar mudanças", "Aprender e se adaptar", "Parar de estudar", "Evitar tecnologia sempre"], c: 1 }
    ];

    let current = 0;
    let score = 0;
    const answers = Array(questions.length).fill(null);

    const questionText = document.getElementById('questionText');
    const optionsContainer = document.getElementById('optionsContainer');
    const questionNumber = document.getElementById('questionNumber');
    const progressBar = document.getElementById('progressBar');
    const prevBtn = document.getElementById('prevBtn');
    const nextBtn = document.getElementById('nextBtn');
    const finishBtn = document.getElementById('finishBtn');
    const scoreInfo = document.getElementById('scoreInfo');
    const quizView = document.getElementById('quizView');
    const resultView = document.getElementById('resultView');
    const finalScore = document.getElementById('finalScore');

    function renderQuestion() {
      const item = questions[current];
      questionNumber.textContent = `Pergunta ${current + 1} de ${questions.length}`;
      questionText.textContent = item.q;
      optionsContainer.innerHTML = '';

      item.o.forEach((opt, index) => {
        const btn = document.createElement('button');
        btn.className = 'option' + (answers[current] === index ? ' selected' : '');
        btn.textContent = opt;
        btn.onclick = () => {
          answers[current] = index;
          updateScore();
          renderQuestion();
        };
        optionsContainer.appendChild(btn);
      });

      progressBar.style.width = `${((current + 1) / questions.length) * 100}%`;
      prevBtn.disabled = current === 0;
      prevBtn.style.opacity = current === 0 ? 0.5 : 1;

      nextBtn.classList.toggle('hidden', current === questions.length - 1);
      finishBtn.classList.toggle('hidden', current !== questions.length - 1);

      scoreInfo.textContent = `Acertos: ${score}`;
    }

    function updateScore() {
      score = 0;
      answers.forEach((ans, i) => {
        if (ans === questions[i].c) score++;
      });
      scoreInfo.textContent = `Acertos: ${score}`;
    }

    prevBtn.onclick = () => {
      if (current > 0) {
        current--;
        renderQuestion();
      }
    };

    nextBtn.onclick = () => {
      if (current < questions.length - 1) {
        current++;
        renderQuestion();
      }
    };

    finishBtn.onclick = () => {
      updateScore();
      quizView.classList.add('hidden');
      resultView.classList.remove('hidden');
      finalScore.textContent = `Você acertou ${score} de ${questions.length} perguntas.`;
    };

    renderQuestion();
  </script>
